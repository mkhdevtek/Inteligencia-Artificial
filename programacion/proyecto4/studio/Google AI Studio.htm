code Python
    from unsloth import FastLanguageModel
import torch
from trl import SFTTrainer
from transformers import TrainingArguments

# 1. Cargar Modelo (Qwen 2.5 Coder 7B) en 4-bit (para que quepa en Colab)
model, tokenizer = FastLanguageModel.from_pretrained(
    model_name = "Qwen/Qwen2.5-Coder-7B-Instruct",
    max_seq_length = 2048,
    load_in_4bit = True,
)

# 2. Configurar PEFT (Parameter Efficient Fine-Tuning)
model = FastLanguageModel.get_peft_model(
    model,
    r = 16, # Rango de los adaptadores
    target_modules = ["q_proj", "k_proj", "v_proj", "o_proj", "gate_proj", "up_proj", "down_proj"],
    lora_alpha = 16,
    lora_dropout = 0,
    bias = "none",
)

# 3. Cargar tu Dataset
from datasets import load_dataset
dataset = load_dataset("json", data_files="tu_dataset_subido.jsonl", split="train")

# 4. Entrenar
trainer = SFTTrainer(
    model = model,
    tokenizer = tokenizer,
    train_dataset = dataset,
    dataset_text_field = "text", # Necesitars formatear tu JSON a texto plano estilo chat
    max_seq_length = 2048,
    args = TrainingArguments(
        per_device_train_batch_size = 2,
        gradient_accumulation_steps = 4,
        warmup_steps = 5,
        max_steps = 60, # Ajustar segn tamao de datos
        learning_rate = 2e-4,
        fp16 = not torch.cuda.is_bf16_supported(),
        bf16 = torch.cuda.is_bf16_supported(),
        output_dir = "outputs",
    ),
)
trainer.train()

# 5. Guardar modelo en formato GGUF (Para usar en tu laptop)
model.save_pretrained_gguf("tutor_algoritmos_gguf", tokenizer, quantization_method = "q4_k_m")

3. Automatizacin (Script en Python)

Si tienes una clave de API de OpenAI (cuesta unos pocos dlares y ahorra das de trabajo), puedes usar este script en tu laptop para generar cientos de ejemplos automticamente.

Guarda esto como generador_dataset.py:

code Python
    import openai
import json
import os

# Configura tu API Key
client = openai.OpenAI(api_key="TU_CLAVE_API_AQUI")

topics = [
    "Bsqueda Binaria", "Ordenamiento MergeSort", "Ordenamiento QuickSort",
    "Listas Enlazadas", "Pilas y Colas", "rboles Binarios",
    "Grafos DFS/BFS", "Recursividad", "Programacin Dinmica",
    "Tablas Hash", "Notacin Big O"
]

def generate_data(topic):
    prompt = f"""
    Genera 5 ejemplos de entrenamiento para un tutor de algoritmos sobre el tema: {topic}.
    Formato: Lista de objetos JSON puros.
    Estructura: {{"instruction": "...", "input": "", "output": "..."}}
    Estilo: El tutor debe explicar paso a paso, ser didctico y usar ejemplos.
    Idioma: Espaol.
    """
    
    try:
        response = client.chat.completions.create(
            model="gpt-4o", # O gpt-3.5-turbo para ahorrar
            messages=[{"role": "system", "content": "Eres un generador de datasets educativos JSON."},
                      {"role": "user", "content": prompt}],
            temperature=0.7
        )
        content = response.choices[0].message.content
        # Limpieza bsica para asegurar formato JSON
        content = content.replace("```json", "").replace("```", "")
        return json.loads(content)
    except Exception as e:
        print(f"Error con {topic}: {e}")
        return []

all_data = []

print("Iniciando generacin de dataset...")
for topic in topics:
    print(f"Generando datos para: {topic}")
    data = generate_data(topic)
    all_data.extend(data)

# Guardar en archivo JSONL
with open("dataset_tutor_algoritmos.jsonl", "w", encoding="utf-8") as f:
    for entry in all_data:
        json.dump(entry, f, ensure_ascii=False)
        f.write("\n")

print(f"Listo! Se generaron {len(all_data)} ejemplos en 'dataset_tutor_algoritmos.jsonl'")
